

import React, { useState, useEffect } from 'react';

export const SecondPage = () => {
    const [startStory, setStartStory] = useState(false);

    useEffect(() => {
        // 5 second delay before the message slides out and canvas slides in
        const timer = setTimeout(() => {
            setStartStory(true);
        }, 5000);

        return () => clearTimeout(timer);
    }, []);

    return (
        <div className="fixed inset-0 bg-slate-950 flex items-center justify-center overflow-hidden">
            
            {/* 1. THE MESSAGE AREA (Slides out after 5s) */}
            <div className={`absolute inset-0 flex flex-col items-center justify-center p-6 transition-all duration-[1500ms] ease-in-out transform ${
                startStory ? '-translate-x-full opacity-0' : 'translate-x-0 opacity-100'
            }`}>
                <div className="text-center space-y-6 max-w-3xl">
                    {/* The Header Message */}
                    <h2 className="text-4xl md:text-6xl font-black text-white tracking-[0.1em] uppercase">
                        Are <span className="text-red-500 animate-pulse">you sure</span> to continue!
                    </h2>
                    
                    {/* The Body Message */}
                    <p className="text-xl md:text-3xl text-slate-300 font-mono italic">
                        "whether you are ready or not <span className="text-cyan-400 font-bold not-italic">let's go.</span>"
                    </p>

                    {/* Simple 5-sec visual countdown bar */}
                    {!startStory && (
                        <div className="w-48 h-1 bg-slate-800 mx-auto mt-8 overflow-hidden rounded-full">
                            <div className="h-full bg-cyan-500 animate-[progress_5s_linear_forwards]"></div>
                        </div>
                    )}
                </div>
            </div>

            {/* 2. THE STORY CANVAS (Slides in after 5s) */}
            <div className={`absolute inset-0 w-full h-full flex items-center justify-center transition-all duration-[1500ms] ease-in-out transform ${
                startStory ? 'translate-x-0' : 'translate-x-full'
            }`}>
                <div className="relative w-[95vw] h-[85vh] bg-slate-900 border-2 border-slate-700 rounded-2xl shadow-2xl overflow-hidden">
                    
                    {/* Inner Grid for the Canvas vibe */}
                    <div className="absolute inset-0 opacity-20 bg-[linear-gradient(to_right,#80808012_1px,transparent_1px),linear-gradient(to_bottom,#80808012_1px,transparent_1px)] bg-[size:40px_40px]"></div>
                    
                    {/* Placeholder for the Story Content / Inputs */}
                    <div className="relative z-10 w-full h-full flex flex-col items-center justify-center">
                        <div className="text-cyan-500 font-mono text-sm tracking-[0.3em] uppercase animate-pulse">
                            [ Story Canvas Active ]
                        </div>
                        {/* Your Character/Story starts here */}
                    </div>

                </div>
            </div>

            {/* Custom Progress Animation for the timer bar */}
            <style jsx>{`
                @keyframes progress {
                    from { width: 0%; }
                    to { width: 100%; }
                }
            `}</style>
        </div>
    );
};
